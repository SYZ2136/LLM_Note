# 15. 三数之和
```
class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        n = len(nums)
        res_set = set()

        # 三重循环：枚举 i, j, k，且 i < j < k
        for i in range(n):  # 从0开始，左闭右开
            for j in range(i + 1, n):  #左闭右开
                for k in range(j + 1, n):
                    if nums[i] + nums[j] + nums[k] == 0:
                        # 排序保证 nums[i] <= nums[j] <= nums[k]
                        res_set.add((nums[i], nums[j], nums[k]))，防止出现(-1, -2, 3) 和 (-2, 1, 3)这种情况，明明应该算作一种，但是加入set还是视为两种，如果排序了，加入set当中就会自动去重

        # set中元素必须hashable，因此其中一定是tuple，最后一步tuple生成list
        return [list(triple) for triple in res_set]

```
不管排不排序，这里三重循环方式枚举出的下标三元组 `(i, j, k)` 遍历且不重复
用n个for循环，类似也能实现 $C_N^n$


```
class Solution:

    def threeSum(self, nums: List[int]) -> List[List[int]]:
        n = len(nums)
        nums.sort()
        s = set()
        for i in range(n):
            lp, rp = i + 1, n - 1
            while lp < rp:
                sum = nums[i] + nums[lp] + nums[rp]
                if sum > 0:
                    rp -= 1
                elif sum < 0:
                    lp += 1
                else:
                    s.add((nums[i], nums[lp], nums[rp]))
                    # 把左右都调到首次非重复位置，极端情况就是lp到rp都相等，二者经过下面两个while之后位置对调
                    while lp < n-1 and nums[lp] == nums[lp+1]:
                        lp+=1
                    while rp > 0 and nums[rp] == nums[rp-1]:
                        rp-=1
                    # 避免无重复时不进上面两个while，随便动一下lp或者rp
                    # 目的就是调到首次不重合的地方，右移lp和左移rp都能实现
                    lp += 1
        return [list(a) for a in s]
```

```python
class Solution:

        count = collections.defaultdict(int)  # 记录 “前缀和为某个数值” 出现的次数
        count[0] = 1
        ans = 0  # 符合要求的子数组个数
        pre = 0  # 前缀和初始值
        for x in nums:
            pre += x
            # pre长 - k = pre短，说明 pre长 - pre短 = k

            ans += count[pre - k]  # defaultdict(int)默认值为0，没出现过的前缀和+0
            count[pre] += 1
        return ans
```

```python
class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        q = deque()  # 队列 q 存的是数组索引，总之存地址信息量更多
        ans = []

        for i, x in enumerate(nums):
            # 窗口从左往右滑，先移除过期元素
            if q and q[0] <= i - k:
                q.popleft()
            # 每个新元素进队，队中比他小的都滚(窗口内比他小的肯定不是最大值，不用维护
            # 从而始终维护双端队列从左到右递减
            while q and nums[q[-1]] <= x:
                q.pop()
            q.append(i)
            # 当形成窗口时，队首(左侧)即为窗口内最大值，取出即可
            if i >= k - 1:
                ans.append(nums[q[0]])
        return ans
```

# 76
先扩右，当满足条件后尽量缩左

人为造一个第三空间计数器 `t_count`
t 中的字符全都 +1
扫 s ，先右扩，如果扫到一样的就 -1，可以理解为一次抑制；因此只要能把对应位置降低到 ≤ 0，说明就覆盖了 t
再左缩，每缩一个，对应位置加一，因为缩一个就释放一个字母，原本的抑制没了；当左侧出现正值，说明这个点确实需要抑制，因此这个点就是开头

```python
class Solution:
    def minWindow(self, s: str, t: str) -> str:
        m = len(s)
        t_count = [0] * 128  # ASCII 码范围是 0-127
        for c in t:
            t_count[ord(c)] += 1
        left = 0	
        ans_left, ans_right = -1, m
        # 开始枚举右边界
        for right in range(m):
            # 加入到窗口中
            t_count[ord(s[right])] -= 1
            # 满足条件的情况下不断缩小左边界
            while all(x <= 0 for x in t_count):
                # 缩小左边界之前先尝试更新答案
                if right - left < ans_right - ans_left:
                    ans_left, ans_right = left, right
                t_count[ord(s[left])] += 1
                left += 1
        return s[ans_left : ans_right + 1] if ans_left >= 0 else ""
```


# 53 
定义 `f[i] = 以 nums[i] 为结尾（包含 nums[i]）的连续子数组最大和`
状态转移方程：当前数组子数组位置的最大值要么是之前的加上当前值，要么是当前值自成一个新段。
```python
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        n = len(nums)
        f = [0] * n
        f[0] = nums[0]
        for i in range(1, n):
            f[i] = max(f[i - 1] + nums[i], nums[i])
        return max(f)
```

正确解法
若 `x ∈ [1, n]`：它的“合法位置”索引为 `x-1`
本质上就是把原本的数组改造成一个线性映射的哈希函数的哈希表，哈希函数就是 x-1

```python
class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:
        n = len(nums)

        # 把所有在[1, n]范围内的元素放到对应位置
        for i in range(n):
            # 这里一定是 while，不能用 if，不然有可能这里是一个[1, n]，后面换过来的一个元素也是[1, n]，如果用 if，那么后面那个换过来的元素就被跳过没被放在正确位置
            while 1 <= nums[i] <= n and nums[nums[i] - 1] != nums[i]:
                nums[nums[i] - 1], nums[i] = nums[i], nums[nums[i] - 1]

        for i in range(n):
            # 不符合这个 if 说明这个元素不在[1, n]，不是合法元素，而且前面的都按照合法放了，说明 [1, n] 到 i+1 这里断掉了，所以返回 i+1
            if nums[i] != i + 1:  
                return i + 1
        
        # 如果 n 个数字分别是 1,..,n
        return n + 1
```

注意 
`nums[nums[i] - 1], nums[i] = nums[i], nums[nums[i] - 1]`
不能写成 
`nums[i], nums[nums[i] - 1] = nums[nums[i] - 1], nums[i]`

O(常数)
先保存第一行第一列是否存在 0 元素的 flag，然后直接把第一行第一列去掉 `matrix[0][0]` 剩下的 m-1 和 n-1 长的数组作为上面 `O(m+n)` 方法中的两个数组存储子矩阵的逻辑

```
class Solution:
    def setZeroes(self, matrix: List[List[int]]) -> None:
        m, n = len(matrix), len(matrix[0])
        # 把“第一行/第一列原本是否需要清零”先保存下来
        flag_col0 = any(matrix[i][0] == 0 for i in range(m))
        flag_row0 = any(matrix[0][j] == 0 for j in range(n))

        for i in range(1, m):
            for j in range(1, n):
                if matrix[i][j] == 0:
                # 这里看似动了第一行第一列内容，但是恰好，如果i,j位置为0，本身第一行第一列对应位置就要置0，这样可以实现用0标记对应行/列有0元素的同时实现修改
                    matrix[i][0] = matrix[0][j] = 0

        for i in range(1, m):
            for j in range(1, n):
                if matrix[i][0] == 0 or matrix[0][j] == 0:
                    matrix[i][j] = 0
                    
        if flag_col0:
            for i in range(m):
                matrix[i][0] = 0

        if flag_row0:
            for j in range(n):
                matrix[0][j] = 0
```

###### 左闭右闭（这个写法错误率低一些）

关键索引就两个部分，一个子矩阵行列索引管理，固定好左开右闭还是左闭右开，只看第一轮代值确定四个部分即可
第二个部分 `ans.append()` 元素索引具体取多少，只看第一圈，用必要条件确定就行，直接代值确定此时添加的行或者列的相关索引是什么即可

```
class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        m, n = len(matrix), len(matrix[0])
        left, right, top, bottom = 0, n - 1, 0, m - 1
        ans = []

        while left <= right and top <= bottom:
            for j in range(left, right + 1):
                ans.append(matrix[top][j])
            top += 1
            if top > bottom:
                break

            for i in range(top, bottom + 1):
                ans.append(matrix[i][right])
            right -= 1
            if left > right:
                break

            for j in range(right, left - 1, -1):
                ans.append(matrix[bottom][j])
            bottom -= 1
            if top > bottom:
                break

            for i in range(bottom, top - 1, -1):
                ans.append(matrix[i][left])
            left += 1

        return ans
```


###### 左闭右开
```
class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        m, n = len(matrix), len(matrix[0])
        # 左闭右开描述初始整个矩阵，所以取索引 m，n
        left, right, top, bottom = 0, n, 0, m
        ans = []

        while left < right and top < bottom:

            for j in range(left, right):
                ans.append(matrix[top][j])
            top += 1
            if top == bottom:  # 这里判断 top >= bottom 也行，因为外圈 while 有限制，边界最多等于，判断 >= 和判断 = 这里是等价的，下面也同理
                break

            for i in range(top, bottom):
                ans.append(matrix[i][right - 1])
            right -= 1
            if left == right:
                break

            for j in range(right - 1, left - 1, -1):
                ans.append(matrix[bottom - 1][j])
            bottom -= 1
            if top == bottom:
                break

            for i in range(bottom - 1, top - 1, -1):
                ans.append(matrix[i][left])
            left += 1

        return ans

```

# 21

思路是用带有头结点的链表风格；
**下面这个 dummy 和 current 的维护方式很重要，**
```
dummy = current = ListNode()
while xxx:
    current.next = list1
    current = current.next
return dummy.next  # dummy 是头结点，所以最终返回值为 dummy.next
```
**dummy 是最终结果的头结点，current 始终指向最终目标的尾部节点**，这是一个标准的代码块通用写法

#### 本体问题解决
dummy 是最终结果的头结点，current 始终指向最终目标的尾部节点，list1 和 list2 在两个输入的链表上滑动遍历；第一次判断 `if list1.val < list2.val:` 时，current.next 改动，对应 dummy.next 也动，因为二者指向同一个对象，但是接着 current 自身就变了，指向新加入的尾端节点；
```python
class ListNode:
    def __init__(self, val: int = 0, next: Optional["ListNode"] = None):
        self.val = val
        self.next = next

class Solution:
    def mergeTwoLists(
        self,
        list1: Optional[ListNode],
        list2: Optional[ListNode],
    ) -> Optional[ListNode]:
        # 不确定头结点指向 list1 还是 list2，所以创建的时候先按照默认值；后面有些题创建的时候就知道了头结点指向，直接创建的时候就可以写 dummy = current = ListNode(next=head)
        dummy = current = ListNode()
        while list1 and list2:
            if list1.val < list2.val:
                current.next = list1
                current = current.next
                list1 = list1.next
            else:
                current.next = list2
                current = current.next
                list2 = list2.next
        current.next = list1 if not list2 else list2
        return dummy.next
```

# 2 两数相加

###### 一种无脑方法，主要锻炼下 python 的一些小的字符串和数字相互转换的语法
```
class Solution:
    def addTwoNumbers(
        self,
        l1: Optional[ListNode],
        l2: Optional[ListNode],
    ) -> Optional[ListNode]:
        dummy = current = ListNode()
        a, b = [], []
        while l1:
            a.append(l1.val)
            l1 = l1.next
        while l2:
            b.append(l2.val)
            l2 = l2.next
        c, d = "", ""
        
        a[:] = a[::-1]
        b[:] = b[::-1]

        for s in a:
            c += str(s)  # 字符串拼接直接是 +， 数字转字符串用 str()
            # 补充： unicode码点和字符串的相互转化用的是 ord() 和 chr()

        for s in b:
            d += str(s)

        outcome_num = int(c) + int(d)
        outcome_str = str(outcome_num)

        outcome_list = [int(x) for x in outcome_str]
        outcome_list[:] = outcome_list[::-1]

        for x in outcome_list:
            p = ListNode(x)
            current.next = p
            current = current.next
```

###### .reverse() 的使用

```
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        dummy = current = ListNode()
        a, b = [], []

        while l1:
            a.append(l1.val)
            l1 = l1.next

        while l2:
            b.append(l2.val)
            l2 = l2.next

        # list 的原地逆转，时间 O(n)，空间 O(1)
        a.reverse()
        b.reverse()

        c = ''.join(str(x) for x in a)
        d = ''.join(str(x) for x in b)

        # 关键修正：不要用 float，会丢精度/出现科学计数法/带 .0
        outcome = str(int(c) + int(d))

        # 题目要求结果链表仍然是逆序存储
        for ch in outcome[::-1]:
            current.next = ListNode(int(ch))
            current = current.next

        return dummy.next
```

###### reversed(), map() 的使用，以及 str 直接切片索引

```
class Solution:
    def addTwoNumbers(
        self,
        l1: Optional[ListNode],
        l2: Optional[ListNode],
```

```
            first = prev.next
            second = pre.next.next
            # 链表节点的逻辑关系仅仅由各个节点的 next 决定，所以，改变三个部分的 .next 值就能实现节点顺序的交换
            prev.next, first.next, second.next = second, second.next, first
            # 此时 first 是调换之后的了，所以 prev = first
            prev = first
        return dummy.next
```

# 反转链表系列

## 206 反转链表

current 表示当前遍历到的节点，pre 表示其 current 前面一个节点；初始位置显然 `pre, cur = None, head`
```
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        # 让 pre = ListNode(),头上就会多出一个默认节点，反转结果就多出来一个节点
        pre, cur = None, head
        # 这一块 while 是标准操作
        while cur:
            nxt = cur.next
            cur.next = pre
            pre = cur
            cur = nxt
        return pre
```

重要结论：
从原来的链表上看：
反转前，
**pre 指向反转这一段的前一个**
**cur 指向反转这一段第一个**
反转结束后
**pre 指向反转这一段的末尾**
**cur 指向反转这一段的后一个**


## 92 反转链表(2)

 上面一道题中是整个链表翻转，反转段处理完了就完成了；本题是反转中间部分，反转段完成之后还要跟前面接上；因此反转段前一个节点引入一个指针 $p_0$，

**总之，反转段前面一个节点标上 $p_0$**

为了避免 left 为第一个节点，第一个节点前面没有节点，所以头上加一个哨兵节点 `dummy = ListNode(next=head)`

```
class Solution:
    def reverseBetween(self, head: Optional[ListNode], left: int, right: int) -> Optional[ListNode]:
        dummy = ListNode(next=head)
        p0 = dummy
        for _ in range(left - 1):
            p0 = p0.next

        # 从这里开始完全就是上一题的逻辑，这里的 pre = None，其实就是把这个选中段倒过来，原本的第一个指向 None，看似断了，但是别急，往后看
        pre = None
        cur = p0.next
        for _ in range(right - left + 1):
            nxt = cur.next
            cur.next = pre
            pre = cur
            cur = nxt

        # 神来之笔，上面那段直接用上一题的逻辑看成一个单独链表的逆转，看似 pre 一开始为 None，倒过来之后这段最后指 None；但是有 p0，p0.next 就是逆转之前的第一个节点，也是逆转之后的最后一个节点；用上一题加粗的结论：从原来的链表上看，反转结束后，pre在反转段的最后一个，current在反转段的后一个(因为是一样的代码，所以自然结论适用)
        p0.next.next = cur
        p0.next = pre
        return dummy.next
```


## 25 K个一组翻转链表

添加一点逻辑就行；先拿到长度，然后 while 循环，单个循环内部的逻辑与 92 一样，只需额外添加一个保存 p0.next 的逻辑，这样能够让 p0 正确到下一段更新的前一个点
```
class Solution:
    def reverseKGroup(self, head: Optional["ListNode"], k: int) -> Optional["ListNode"]:
        p0 = dummy = ListNode(next=head)
        n = 0
        cur = head
        while cur:
            n += 1
            cur = cur.next

        
        while n >= k:
            n -= k

            pre = None
            cur = p0.next
      
            for _ in range(k):
                nxt = cur.next
                cur.next = pre
                pre = cur
                cur = nxt

          
            temp = p0.next        
            temp.next = cur        
            p0.next = pre          
            p0 = temp              
            
        return dummy.next

```

# 114. 二叉树展开链表

要求head 初始为 None，然后原地操作，还用 TreeNode，因此最后还是一颗树，head 指向根，所以用头插法；

头插法就是，初始 head 空，一个 node 进来指向 head，然后 head 指向这个 node

先序是根左右，但是不能从根开始就让左边为空，会丢信息；所以倒过来，用右左根，遍历到"根的时候" 进行变形操作；
想象一下，**第一个退栈的节点是叶子，左右都没子树**，这样正好头插到 head 上


```python
class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        head = None
        def dfs(node: TreeNode | None) -> None:  
            if node is None:
                return
            nonlocal head
            dfs(node.right)
            dfs(node.left)
            node.right = head
            node.left = None
            head = node
        dfs(root)
```


理解递归的过程，大脑模拟复杂情况不现实，就用特例帮助理解写代码就行；

```
         1
        / \
       2   3
          / \
         4   5
```

就按下面的 右左根的特殊后序遍历 特例去想栈的变化情况就行
```
dfs(root.right)
dfs(root.left)
root # 进行某种操作
```

关于其中的 "root # 进行某种操作" ，不要关注名字，就是一个当前情况，root 表示假设当前递推位于的位置叫做 root，对他进行什么操作，**因此具体写法我们看边界条件**，也就相当于数学归纳法的 a0 位置往上推的操作是什么；**比如上面这题的展开链表；操作就是** 
```
    node.right = head
    node.left = None
    head = node
```
**写法就是，如果当前 node 在最后一个叶子，如何头插到 head** ，这样就非常好写了

说白了就是用特殊情况帮忙把代码写出来就行；

# 98. 验证 BST

第二种解法：

```python
class Solution:
    pre = -inf
    def isValidBST(self, root: Optional[TreeNode], left: float = -inf, right: float = inf) -> bool:
        if root is None:
            return True
        if not self.isValidBST(root.left):
            return False
        if root.val <= self.pre:
            return False
        self.pre = root.val
        return self.isValidBST(root.right)
```
