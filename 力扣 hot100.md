# 15. 三数之和
```
class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        n = len(nums)
        res_set = set()

        # 三重循环：枚举 i, j, k，且 i < j < k
        for i in range(n):  # 从0开始，左闭右开
            for j in range(i + 1, n):  #左闭右开
                for k in range(j + 1, n):
                    if nums[i] + nums[j] + nums[k] == 0:
                        # 排序保证 nums[i] <= nums[j] <= nums[k]
                        res_set.add((nums[i], nums[j], nums[k]))，防止出现(-1, -2, 3) 和 (-2, 1, 3)这种情况，明明应该算作一种，但是加入set还是视为两种，如果排序了，加入set当中就会自动去重

        # set中元素必须hashable，因此其中一定是tuple，最后一步tuple生成list
        return [list(triple) for triple in res_set]

```
不管排不排序，这里三重循环方式枚举出的下标三元组 `(i, j, k)` 遍历且不重复
用n个for循环，类似也能实现 $C_N^n$


```
class Solution:

    def threeSum(self, nums: List[int]) -> List[List[int]]:
        n = len(nums)
        nums.sort()
        s = set()
        for i in range(n):
            lp, rp = i + 1, n - 1
            while lp < rp:
                sum = nums[i] + nums[lp] + nums[rp]
                if sum > 0:
                    rp -= 1
                elif sum < 0:
                    lp += 1
                else:
                    s.add((nums[i], nums[lp], nums[rp]))
                    # 把左右都调到首次非重复位置，极端情况就是lp到rp都相等，二者经过下面两个while之后位置对调
                    while lp < n-1 and nums[lp] == nums[lp+1]:
                        lp+=1
                    while rp > 0 and nums[rp] == nums[rp-1]:
                        rp-=1
                    # 避免无重复时不进上面两个while，随便动一下lp或者rp
                    # 目的就是调到首次不重合的地方，右移lp和左移rp都能实现
                    lp += 1
        return [list(a) for a in s]
```

```python
class Solution:

        count = collections.defaultdict(int)  # 记录 “前缀和为某个数值” 出现的次数
        count[0] = 1
        ans = 0  # 符合要求的子数组个数
```

```python
class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        q = deque()  # 队列 q 存的是数组索引，总之存地址信息量更多
        ans = []

        for i, x in enumerate(nums):
            # 窗口从左往右滑，先移除过期元素
            if q and q[0] <= i - k:
                q.popleft()
            # 每个新元素进队，队中比他小的都滚(窗口内比他小的肯定不是最大值，不用维护
            # 从而始终维护双端队列从左到右递减
            while q and nums[q[-1]] <= x:
                q.pop()
            q.append(i)
            # 当形成窗口时，队首(左侧)即为窗口内最大值，取出即可
            if i >= k - 1:
                ans.append(nums[q[0]])
        return ans
```

# 76
先扩右，当满足条件后尽量缩左

人为造一个第三空间计数器 `t_count`
t 中的字符全都 +1
扫 s ，先右扩，如果扫到一样的就 -1，可以理解为一次抑制；因此只要能把对应位置降低到 ≤ 0，说明就覆盖了 t
再左缩，每缩一个，对应位置加一，因为缩一个就释放一个字母，原本的抑制没了；当左侧出现正值，说明这个点确实需要抑制，因此这个点就是开头

```python
class Solution:
    def minWindow(self, s: str, t: str) -> str:
        m = len(s)
        t_count = [0] * 128  # ASCII 码范围是 0-127
        for c in t:
            t_count[ord(c)] += 1
        left = 0	
        ans_left, ans_right = -1, m
        # 开始枚举右边界
        for right in range(m):
            # 加入到窗口中
            t_count[ord(s[right])] -= 1
            # 满足条件的情况下不断缩小左边界
            while all(x <= 0 for x in t_count):
                # 缩小左边界之前先尝试更新答案
                if right - left < ans_right - ans_left:
                    ans_left, ans_right = left, right
                t_count[ord(s[left])] += 1
                left += 1
        return s[ans_left : ans_right + 1] if ans_left >= 0 else ""
```


# 53 
定义 `f[i] = 以 nums[i] 为结尾（包含 nums[i]）的连续子数组最大和`
状态转移方程：当前数组子数组位置的最大值要么是之前的加上当前值，要么是当前值自成一个新段。
```python
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        n = len(nums)
        f = [0] * n
        f[0] = nums[0]
        for i in range(1, n):
            f[i] = max(f[i - 1] + nums[i], nums[i])
        return max(f)
```

正确解法
若 `x ∈ [1, n]`：它的“合法位置”索引为 `x-1`
本质上就是把原本的数组改造成一个线性映射的哈希函数的哈希表，哈希函数就是 x-1

```python
class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:
        n = len(nums)

        # 把所有在[1, n]范围内的元素放到对应位置
        for i in range(n):
            # 这里一定是 while，不能用 if，不然有可能这里是一个[1, n]，后面换过来的一个元素也是[1, n]，如果用 if，那么后面那个换过来的元素就被跳过没被放在正确位置
            while 1 <= nums[i] <= n and nums[nums[i] - 1] != nums[i]:
                nums[nums[i] - 1], nums[i] = nums[i], nums[nums[i] - 1]

        for i in range(n):
            # 不符合这个 if 说明这个元素不在[1, n]，不是合法元素，而且前面的都按照合法放了，说明 [1, n] 到 i+1 这里断掉了，所以返回 i+1
            if nums[i] != i + 1:  
                return i + 1
        
        # 如果 n 个数字分别是 1,..,n
        return n + 1
```

注意 
`nums[nums[i] - 1], nums[i] = nums[i], nums[nums[i] - 1]`
不能写成 
`nums[i], nums[nums[i] - 1] = nums[nums[i] - 1], nums[i]`

O(常数)
先保存第一行第一列是否存在 0 元素的 flag，然后直接把第一行第一列去掉 `matrix[0][0]` 剩下的 m-1 和 n-1 长的数组作为上面 `O(m+n)` 方法中的两个数组存储子矩阵的逻辑

```
class Solution:
    def setZeroes(self, matrix: List[List[int]]) -> None:
        m, n = len(matrix), len(matrix[0])
        # 把“第一行/第一列原本是否需要清零”先保存下来
        flag_col0 = any(matrix[i][0] == 0 for i in range(m))
        flag_row0 = any(matrix[0][j] == 0 for j in range(n))

        for i in range(1, m):
            for j in range(1, n):
                if matrix[i][j] == 0:
                # 这里看似动了第一行第一列内容，但是恰好，如果i,j位置为0，本身第一行第一列对应位置就要置0，这样可以实现用0标记对应行/列有0元素的同时实现修改
                    matrix[i][0] = matrix[0][j] = 0

        for i in range(1, m):
            for j in range(1, n):
                if matrix[i][0] == 0 or matrix[0][j] == 0:
                    matrix[i][j] = 0
                    
        if flag_col0:
            for i in range(m):
                matrix[i][0] = 0

        if flag_row0:
            for j in range(n):
                matrix[0][j] = 0
```

###### 左闭右闭（这个写法错误率低一些）

关键索引就两个部分，一个子矩阵行列索引管理，固定好左开右闭还是左闭右开，只看第一轮代值确定四个部分即可
第二个部分 `ans.append()` 元素索引具体取多少，只看第一圈，用必要条件确定就行，直接代值确定此时添加的行或者列的相关索引是什么即可

```
class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        m, n = len(matrix), len(matrix[0])
        left, right, top, bottom = 0, n - 1, 0, m - 1
        ans = []

        while left <= right and top <= bottom:
            for j in range(left, right + 1):
                ans.append(matrix[top][j])
            top += 1
            if top > bottom:
                break

            for i in range(top, bottom + 1):
                ans.append(matrix[i][right])
            right -= 1
            if left > right:
                break

            for j in range(right, left - 1, -1):
                ans.append(matrix[bottom][j])
            bottom -= 1
            if top > bottom:
                break

            for i in range(bottom, top - 1, -1):
                ans.append(matrix[i][left])
            left += 1

        return ans
```


###### 左闭右开
```
class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        m, n = len(matrix), len(matrix[0])
        # 左闭右开描述初始整个矩阵，所以取索引 m，n
        left, right, top, bottom = 0, n, 0, m
        ans = []

        while left < right and top < bottom:

            for j in range(left, right):
                ans.append(matrix[top][j])
            top += 1
            if top == bottom:  # 这里判断 top >= bottom 也行，因为外圈 while 有限制，边界最多等于，判断 >= 和判断 = 这里是等价的，下面也同理
                break

            for i in range(top, bottom):
                ans.append(matrix[i][right - 1])
            right -= 1
            if left == right:
                break

            for j in range(right - 1, left - 1, -1):
                ans.append(matrix[bottom - 1][j])
            bottom -= 1
            if top == bottom:
                break

            for i in range(bottom - 1, top - 1, -1):
                ans.append(matrix[i][left])
            left += 1

        return ans

```

# 21

思路是用带有头结点的链表风格；
**下面这个 dummy 和 current 的维护方式很重要，**
```
dummy = current = ListNode()
while xxx:
    current.next = list1
    current = current.next
return dummy.next  # dummy 是头结点，所以最终返回值为 dummy.next
```
**dummy 是最终结果的头结点，current 始终指向最终目标的尾部节点**，这是一个标准的代码块通用写法

#### 本体问题解决
dummy 是最终结果的头结点，current 始终指向最终目标的尾部节点，list1 和 list2 在两个输入的链表上滑动遍历；第一次判断 `if list1.val < list2.val:` 时，current.next 改动，对应 dummy.next 也动，因为二者指向同一个对象，但是接着 current 自身就变了，指向新加入的尾端节点；
```python
class ListNode:
    def __init__(self, val: int = 0, next: Optional["ListNode"] = None):
        self.val = val
        self.next = next

class Solution:
    def mergeTwoLists(
        self,
        list1: Optional[ListNode],
        list2: Optional[ListNode],
    ) -> Optional[ListNode]:
        # 不确定头结点指向 list1 还是 list2，所以创建的时候先按照默认值；后面有些题创建的时候就知道了头结点指向，直接创建的时候就可以写 dummy = current = ListNode(next=head)
        dummy = current = ListNode()
        while list1 and list2:
            if list1.val < list2.val:
                current.next = list1
                current = current.next
                list1 = list1.next
            else:
                current.next = list2
                current = current.next
                list2 = list2.next
        current.next = list1 if not list2 else list2
        return dummy.next
```

# 2 两数相加

###### 一种无脑方法，主要锻炼下 python 的一些小的字符串和数字相互转换的语法
```
class Solution:
    def addTwoNumbers(
        self,
        l1: Optional[ListNode],
        l2: Optional[ListNode],
    ) -> Optional[ListNode]:
        dummy = current = ListNode()
        a, b = [], []
        while l1:
            a.append(l1.val)
            l1 = l1.next
        while l2:
            b.append(l2.val)
            l2 = l2.next
        c, d = "", ""
        
        a[:] = a[::-1]
        b[:] = b[::-1]

        for s in a:
            c += str(s)  # 字符串拼接直接是 +， 数字转字符串用 str()

            d += str(s)

        outcome_num = int(c) + int(d)
        outcome_str = str(outcome_num)

        outcome_list = [int(x) for x in outcome_str]
        outcome_list[:] = outcome_list[::-1]

            p = ListNode(x)
            current.next = p
```

###### .reverse() 的使用

```
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        dummy = current = ListNode()
        a, b = [], []

        while l1:
            a.append(l1.val)
            l1 = l1.next

        while l2:
            b.append(l2.val)
            l2 = l2.next

        # list 的原地逆转，时间 O(n)，空间 O(1)
        a.reverse()
        b.reverse()

        c = ''.join(str(x) for x in a)
        d = ''.join(str(x) for x in b)

        # 关键修正：不要用 float，会丢精度/出现科学计数法/带 .0
        outcome = str(int(c) + int(d))

        # 题目要求结果链表仍然是逆序存储
        for ch in outcome[::-1]:
            current.next = ListNode(int(ch))
            current = current.next

        return dummy.next
```

```

            first = prev.next
            second = pre.next.next
            # 链表节点的逻辑关系仅仅由各个节点的 next 决定，所以，改变三个部分的 .next 值就能实现节点顺序的交换
            prev.next, first.next, second.next = second, second.next, first
            # 此时 first 是调换之后的了，所以 prev = first
            prev = first
        return dummy.next
```

## 206 反转链表

current 表示当前遍历到的节点，pre 表示其 current 前面一个节点；初始位置显然 `pre, cur = None, head`
```
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        # 让 pre = ListNode(),头上就会多出一个默认节点，反转结果就多出来一个节点
        pre, cur = None, head

            pre = cur
            cur = nxt
```

反转前，
**pre 指向反转这一段的前一个**
**cur 指向反转这一段第一个**

**pre 指向反转这一段的末尾**
**cur 指向反转这一段的后一个**

```
        p0 = dummy = ListNode(next=head)
        n = 0
        cur = head

            n += 1
            cur = cur.next

        
        while n >= k:
            n -= k

            cur = p0.next

                cur.next = pre
                pre = cur
                cur = nxt

          
            temp = p0.next        
            temp.next = cur        
            p0.next = pre          
            p0 = temp              
            
        return dummy.next
```

#  994. 橘子

这题要做的是 **按层 BFS**，一层代表一分钟。  
`tmp = q` 的作用，是把“这一分钟开始时已经腐烂的橘子”固定下来，只让它们在这一分钟传播。  
新感染的橘子要放进新的 `q`，等 **下一分钟** 再继续传播。


``` python
class Solution:
    def orangesRotting(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        rot = []
        fresh = 0

        for i, row in enumerate(grid):
            for j, x in enumerate(row):
                if x == 1:
                    fresh += 1
                elif x == 2:
                    rot.append((i, j))

        ans = 0
        # 同时有腐烂的和新鲜的才有可能扩散
        while rot and fresh:
            ans += 1
            # 以第一次进去为例，rot 就是 t=0 时腐烂的东西，temp 记录这一圈的，然后 rot 置[], 然后 rot 用于记录下一个时间步扩散的 腐烂果子；这样相当于 rot 每次记录一圈，这样一圈一圈往外 BFS
            temp = rot
            rot = []

            for x, y in temp:
                for i, j in (x - 1, y), (x + 1, y), (x, y - 1), (x, y + 1):
                    if 0 <= i < m and 0 <= j < n and grid[i][j] == 1:
                        rot.append((i, j))
                        fresh -= 1
                        grid[i][j] = 2
         
        # 如果最后还有新鲜的，说明新鲜的是个孤岛，不会被扩散
        return -1 if fresh else ans
```

```python
class Solution:
    def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
        g = [[] for _ in range(numCourses)]
        for a, b in prerequisites:
            g[b].append(a)

        colors = [0] * numCourses

        def dfs(x: int) -> bool:
            colors[x] = 1
            for y in g[x]:
                if colors[y] == 1 or (colors[y] == 0 and dfs(y)):
                    return True
            colors[x] = 2
            return False

        for i in range(numCourses):
            if colors[i] == 0 and dfs(i):
                return False

        return True
```

# 46. 全排列


回溯 可以解决：
组合，排列，切割，子集

**回溯模版：**
```python
def backtrack(状态参数):
    # 1. 终止条件
    if 满足结束条件:
        收集结果
        return

    # 2. 枚举当前层的所有选择
    for 选择 in 当前可选项:
        在当前层做选择
        递归进入下一层，backtrack(下一层状态)
        返回当前层尝试别的结果(回溯)
```


本题的解决过程是一个回溯，也可以理解为一个 dfs，
回溯是：选择，撤销，再选择
dfs 是：一条路一直走到头，然后返回上一层，再往下走


```python
class Solution:
    def permute(self, nums: list[int]) -> list[list[int]]:
        ans: list[list[int]] = []
        path: list[int] = []

        def dfs(s: set[int]) -> None:
            if not s:
                ans.append(path.copy())
                return
            # 比如set([1, 2 ,3]) 的全排列；第一次第一次进入 for 选的 x=1 是第一个元素，也就是排列起点，又进一个 dfs 栈，直到堆到 s 为空，那么尾部退栈，开始选新的尾巴, 过程类似 [1, 2, 3], [1, 3, 2], 然后以 1 为起点结束； x = 1 的所有 dfs 退栈退完了之后，换到下一个 x =2 ，[2, 1, 3], [2, 3, 1]；
            for x in s:
                path.append(x)         # 做选择
                backtrack(s - {x})     # 递归到下一层
                path.pop()             # 撤销选择

        dfs(set(nums))
        return ans
```

# 78. 子集


上一题全排列，每次的状态是从剩余的集合中再挑一个元素
本题子集，每个索引位置的数有 "选" 和 "不选" 两种状态，这里的递归 dfs()表示我现在走到第 `i` 个元素了，选还是不选；

到了第 i 个元素，我有两种合法的选择，如果选了...., 如果不选....， 这两种是分叉选择，不是先后依赖，可调换顺序

```
# 如果选了当前 i，那么此次 path 加入这个元素之后，再进入下一个的栈
path.append(nums[i])
dfs(i + 1)
path.pop()

# 如果不选当前 i ，就进入下一个的栈
dfs(i + 1)
```

后半段调换顺序，按照下面的顺序也可:
```
    dfs(i + 1)  # 考虑下一个数 nums[i+1] 选或不选
    
    path.append(nums[i])
    dfs(i + 1)  # 考虑下一个数 nums[i+1] 选或不选
     path.pop()  # 恢复现场，撤销 path.append(nums[i])
    

```


如果选了 i，那么dfs(i+1）退栈后，自然要考虑不选的情况，因此要把这个 i pop掉
回到进入这条分支之前的状态。

```python
from typing import List

class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        n = len(nums)
        ans = []
        path = []

        # 选或不选：讨论 nums[i] 是否加入 path
        def dfs(i: int) -> None:
            # 最后一个元素选或不选都考虑完了，说明此次子集构造完了
            if i == n:  
                ans.append(path.copy())  # 复制 path，也可以写 path[:]
                return

            # 选 nums[i]
            path.append(nums[i])
            dfs(i + 1)  # 考虑下一个数 nums[i+1] 选或不选
            path.pop()  # 恢复现场，撤销 path.append(nums[i])
            
            # 如果不选 nums[i]
            dfs(i + 1)  # 考虑下一个数 nums[i+1] 选或不选

        dfs(0)
        return ans
```

# 34. 排序数组中找 target 的起始和终止位置

#### 普通的二分查找
首先很熟悉的是**普通的二分查找的闭区间**写法代码:
```python
class Solution:  # 返回目标值的索引
    def binarySearch(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1  # 闭区间 [left, right]
        while left <= right:
            mid = (left + right) // 2
            if target == nums[mid] :
                return mid
            elif target < nums[mid]:
                left = mid + 1
            else:
                right = mid - 1
        return -1
```
当数组里有多个目标值时，`mid` 落在哪个相同值上，取决于当时区间怎么缩小，位置不确定；


#### 一般情况
**这道题是一般的情况**，既然是一般的情况，当然可以用上面的特殊情况辅助我们写一般情况的代码；**精确找到的前提是找到**

核心思路就是定义一个 `lower_bound`，找到 >= target 的第一个元素；
其余的情况都可以转换为 `lower_bound`，(*具体见 灵神 二分查找红蓝染色法 视频)*

```python
def lower_bound(nums: List[int], target: int) -> int:
    left, right = 0, len(nums) - 1  # 闭区间 [left, right]
    while left <= right:  
        mid = (left + right) // 2
        if target < nums[mid]:
            left = mid + 1  
        elif target > nums[mid]:
            right = mid - 1  
        else:
            right = mid - 1
    return left
```

#### 前半段与后半段的理解方式
 
 mid 是边界定位器，用于判断区间如何缩小；想象一下一个数组，找第一个 >= target 的位置，那么 target 在目标数组中是一个固定存在的位置，mid 是一个锚定量，逼近这个实际的 target 位置，按照上面普通二分查找的理解，最终 mid 就是目标值，所以可以理解为最终 mid 目标就是落在 target 实际位置上，这样帮助我们写代码判断边界条件，
 
 **数组分为两段，前半段都是 < target, 后半段都是 >= target，目的是找后半段的第一个元素**
因此可以一次性看清 if 分支，给出下面这种：
```python
class Solution:
    def lower_bound(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1  
        while left <= right:  
            mid = (left + right) // 2
            if nums[mid] >= target:  # 说明此时 mid 还在后半段，也就是目标点 target 的右边，因此缩右让 mid 靠近 target
                right = mid - 1 
            elif nums[mid] < target:  # 说明 mid 在目标点左侧
                left = mid + 1  
        return left
```

**注意上面只要 `nums[mid] >= target`，包括取等，就可以说明 mid 当前在右半段(>=target) 的那段**


#### 完整解答
这里是一般情况，想找到 target 的第一个位置，那么肯定也符合找到了 target，也就是符合上面上面特殊情况二分查找，`nums[mid] < target` 和 `nums[mid] > target` 肯定还是一样的逻辑，因为这两种连目标值都没有；
`nums[mid] = target` 为了精确到第一个位置，不再直接返回 mid，想象一个特例，8,8,8，L 和 R指向第一个和第三个8，那么 mid 指向第二个8，">= target 的第一个元素" 还在左边，因此返回 mid -1;


**维护的闭区间中含有 ">= target 的第一个元素"**
```
[ < target ] [ 维护的区间 ] [ >= target ]
                [L, R]
```
始终满足循环不变量：
```
L - 1 < target
R + 1 >= target
```

**最终一定有 L = R + 1, 因此最后返回 left**

还是用特殊的普通二分查找理解下面的东西：
如果 target < 全数组，说明 target 一直在左半边，所以 left 一直没动, left 为初始值 0 
如果 target > 全数组，说明 target 一直在右半边，所以 right 一直没动，right 为初始值 len(nums) - 1, 而终止条件始终有 L = R + 1，可推得最终返回 left = len(nums)



最终完整解法：
```python
class Solution:
    def lower_bound(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1  
        while left <= right:  
            mid = (left + right) // 2
            if nums[mid] >= target:
                right = mid - 1 
            else:
                left = mid + 1  
        return left

    def searchRange(self, nums: List[int], target: int) -> List[int]:
        start = self.lower_bound(nums, target)
        end = self.lower_bound(nums, target + 1) - 1
        # 不存在的情况有：全大于，全小于，目标值在数组区间里但是不在数组里，最终只可能返回的 left 在数组索引里面，或者 left == len(nums)，因此以下两个条件即可涵盖
        if start == len(nums) or nums[start] != target:  # or 是短路求值
            return [-1, -1]

        return [start, end]
```

# 33. 搜索旋转排序数组

注意数组中元素各不相同

#### 前半段与后半段的理解方式
跟 34 中的 `lower_bound` 一个思路，目的是找到第一个 `nums[i] <= nums[-1]` 的索引 i;

因为原本有序的数组旋转 (原本数组头尾相连，然后顺时针旋转，原本水平的这一段展示出的数字作为旋转后的结果，旋转之后 前半段元素 > `nums[-1]`， 后半段元素 <= `nums[-1]`

注意下面 lower_bound 只是找到第一个 >= 的位置，找没找到 target 还要额外判断

```python
class Solution:
    def findmin(self, nums: list[int]) -> int:
        left, right = 0, len(nums) - 1
        # 找第一个 nums[i] <= nums[-1] 的索引
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] <= nums[-1]:
                right = mid - 1
            else:
                left = mid + 1
        return left

    def lower_bound(self, nums: list[int], left: int, right: int, target) -> int:
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] >= target:
                right = mid - 1
            else:
                left = mid + 1
        return left

    def search(self, nums: List[int], target: int) -> int:
        k = self.findmin(nums)

        # 搜到有两种情况，左半边或者右半边
        # 搜到，在左半边的情况
        if target > nums[-1]:
            idx = self.lower_bound(nums, 0, k - 1, target)
            if not (idx == k or nums[idx] != target):
                return idx

        # 右半边搜到
        elif target == nums[-1]:
            return len(nums) - 1

        elif target < nums[-1]:
            idx = self.lower_bound(nums, k, len(nums) - 1, target)
            # 借用 34 里面没搜到的条件判断再取个反
            if not (idx == len(nums) or nums[idx] != target):
                return idx

        # 剩下就是没搜到的情况
        return -1
```

# 153. 旋转数组最小值

目标逼近位置是首次 `nums[i] <= nums[-1]`，这个位置可以想象为中间某段，这跟 34 就一个思路了，都是锚点为中间某个点，包含这个点往右为后半段，前面的为前半段

**因此只要中途判断 `if nums[mid] <= nums[-1]` ，小于和等于的情况都是后半段，前后半段的分界点是目标位置，而 mid 要逼近目标位置，所以此时 mid 在目标位置右边，自然 `right = mid - 1`**

```python
class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1

        while left <= right:
            mid = (left + right) // 2
            if nums[mid] <= nums[-1]:
                right = mid - 1
            else:
                left = mid + 1

        return nums[left]
```
